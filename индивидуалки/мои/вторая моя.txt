import java.util.Scanner;

// Интерфейс Integer
import java.util.*;
interface MyInteger {
    MyInteger add(MyInteger x);
    MyInteger sub(MyInteger x);
    void print();
}


// Класс Decimal (десятичное число)
class Decimal implements MyInteger {
    int[] digits;   // массив цифр
    int value;      // числовое значение

    Decimal(int value) {
        this.value = value;
        String s = String.valueOf(value);
        digits = new int[s.length()];
        for (int i = 0; i < s.length(); i++) {
            digits[i] = s.charAt(i) - '0';
        }
    }

    public MyInteger add(MyInteger x) {
        Decimal d = (Decimal) x;
        return new Decimal(this.value + d.value);
    }

    public MyInteger sub(MyInteger x) {
        Decimal d = (Decimal) x;
        return new Decimal(this.value - d.value);
    }

    public void print() {
        System.out.print("Decimal: ");
        for (int d : digits)
            System.out.print(d);
        System.out.println();
    }
}

// Класс Binary (двоичное число)
import java.util.*;

class Binary implements MyInteger {
    int[] digits;
    int value;

    Binary(String bin) {
        digits = new int[bin.length()];
        for (int i = 0; i < bin.length(); i++) {
            digits[i] = bin.charAt(i) - '0';
        }

        value = java.lang.Integer.parseInt(bin, 2);
    }

    public MyInteger add(MyInteger x) {
        Binary b = (Binary) x;
        int res = this.value + b.value;

        return new Binary(java.lang.Integer.toBinaryString(res));
    }

    public MyInteger sub(MyInteger x) {
        Binary b = (Binary) x;
        int res = this.value - b.value;

        return new Binary(java.lang.Integer.toBinaryString(res));
    }

    public void print() {
        System.out.print("Binary: ");
        for (int d : digits)
            System.out.print(d);
        System.out.println();
    }
}


// Главный класс
public class InterfaceInteger {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        MyInteger[] arr;

        System.out.print("Vvedite kol-vo elementov: ");
        int k = sc.nextInt();
        arr = new MyInteger[k];

        for (int i = 0; i < k; i++) {
            System.out.println("1 - Decimal, 2 - Binary");
            int p = sc.nextInt();

            if (p == 1) {
                System.out.print("Vvedite desyatichnoe cislo: ");
                int v = sc.nextInt();
                arr[i] = new Decimal(v);
            }

            if (p == 2) {
                System.out.print("Vvedite dvoichnoe cislo: ");
                String b = sc.next();
                arr[i] = new Binary(b);
            }
        }

        System.out.println("\nVivod elementov:");
        for (int i = 0; i < k; i++) {
            arr[i].print();
        }
    }
}
